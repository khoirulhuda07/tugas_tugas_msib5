//Buat table produk
CREATE TABLE produk (id int AUTO_INCREMENT PRIMARY KEY, kode varchar(10), nama varchar(45), harga_beli double, harga_jual double, stock int, min_stok int, jenis_produk_id int not null REFERENCES jenis_produk(id));
//Buat table pesanan_items
CREATE TABLE jenis_items (id int AUTO_INCREMENT PRIMARY key, produk_id int, pesanan_id int, qty int, harga double, FOREIGN KEY (produk_id) REFERENCES produk(id), FOREIGN KEY (pesanan_id) REFERENCES pesanan(id));
//Buat table vendor
CREATE TABLE vendor (id int AUTO_INCREMENT PRIMARY KEY, nomor varchar(10), nama varchar(40), kota varchar(40), kontak varchar(20));
//Buat table Pembelian
CREATE TABLE pembelian (id int AUTO_INCREMENT PRIMARY KEY, tanggal date, nomor varchar(10), produk_id int, jumlah int, harga double, vendor_id int, FOREIGN KEY (produk_id) REFERENCES produk(id), FOREIGN KEY (vendor_id) REFERENCES vendor(id));
